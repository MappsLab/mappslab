# Mapp

The Mapp component renders the map.

<MapWrapper>
	<Mapp
		APIKey="AIzaSyCOqxjWmEzFlHKC9w-iUZ5zL2rIyBglAag"
		options={{
			zoom: 9
		}}
		render={({ map }) => (
			pins.map(({lat, lang, ...rest}) => (
				<Mapp.Marker
					key={rest.uid}
					position={{
						lat: lat,
						lng: lang
					}}
					onClick={() => {
						alert(`Clicked pin ${rest.title}`)
					}}
					{...rest}
					map={map}
				/>
			)
		))}
	/>
</MapWrapper>
